import '../styles.css'
import Head from 'next/head'
import { useEffect } from 'react'
import { redirect } from 'next/navigation'

export default function App({ Component, pageProps }) {
    useEffect(() => {
      if (typeof window === 'undefined') {
        return
      }

      const { href } = window.location
      if (href.includes('/docs/main/doc')) {
        // retro-compatibility with old URLs that used to have /doc in them before we merged components and main docs
        window.location.href = href.replace('/docs/main/doc', '/docs/main')
      }
  }, [])

  return (
    <>
      <Head>
        <link rel="icon" type="image/x-icon" href="/assets/images/logo/platon.svg" />
      </Head>
      <Component {...pageProps} />
    </>
  )
}
